import { Component } from '@angular/core'; @Component({ selector: 'app-carousel', templateUrl: './carousel.component.html', styleUrls: ['./carousel.component.css'] }) export class CarouselComponent { currentSlide = 0; carouselData: any[] = []; // Replace 'any[]' with the actual type of your carousel items // Method to navigate to the previous slide prevSlide() { this.currentSlide = (this.currentSlide - 1 + this.carouselData.length) % this.carouselData.length; } // Method to navigate to the next slide nextSlide() { this.currentSlide = (this.currentSlide + 1) % this.carouselData.length; } }
